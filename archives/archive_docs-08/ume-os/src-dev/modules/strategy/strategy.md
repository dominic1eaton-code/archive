use actix_web::{web, App, HttpServer, HttpResponse, Responder};
use shared_lib::workflow::{WorkflowTask, TaskStatus};
use shared_lib::rabbitmq;
use serde_json::json;
use std::collections::HashMap;
use tokio::sync::Mutex;
use std::sync::Arc;

type TaskMap = Arc<Mutex<HashMap<String, WorkflowTask>>>;

#[tokio::main]
async fn main() -> std::io::Result<()> {
    let tasks: TaskMap = Arc::new(Mutex::new(HashMap::new()));

    // Example: start RabbitMQ listener in background
    let tasks_clone = tasks.clone();
    tokio::spawn(async move {
        let channel = rabbitmq::create_channel("amqp://guest:guest@localhost:5672").await;
        channel.queue_declare("workflow_tasks", Default::default(), Default::default()).await.unwrap();
        let mut consumer = channel.basic_consume("workflow_tasks", "workflow_service", Default::default(), Default::default()).await.unwrap();

        while let Some(delivery) = consumer.next().await {
            let (_, delivery) = delivery.unwrap();
            let task: WorkflowTask = serde_json::from_slice(&delivery.data).unwrap();
            println!("Received workflow task: {:?}", task.task_name);

            let mut task_map = tasks_clone.lock().await;
            task_map.insert(task.id.to_string(), task.clone());

            delivery.ack(Default::default()).await.unwrap();
        }
    });

    // API endpoint to trigger workflow execution
    let tasks_clone2 = tasks.clone();
    async fn execute_workflows(tasks: web::Data<TaskMap>) -> impl Responder {
        let mut task_map = tasks.lock().await;

        for (_id, task) in task_map.iter_mut() {
            if task.status == TaskStatus::Pending && task.dependencies.iter().all(|dep| {
                task_map.get(&dep.to_string()).map_or(true, |t| t.status == TaskStatus::Completed)
            }) {
                println!("Executing task: {}", task.task_name);
                task.status = TaskStatus::InProgress;

                // Simulate execution
                tokio::time::sleep(std::time::Duration::from_secs(1)).await;

                // Random success/failure simulation
                if rand::random::<f32>() > 0.2 {
                    task.status = TaskStatus::Completed;
                    println!("Task '{}' completed", task.task_name);
                } else {
                    task.retries += 1;
                    task.status = if task.retries <= task.max_retries { TaskStatus::Pending } else { TaskStatus::Failed };
                    println!("Task '{}' failed, retry {}/{}", task.task_name, task.retries, task.max_retries);
                }
            }
        }

        HttpResponse::Ok().json(json!({"status": "workflow execution cycle complete"}))
    }

    HttpServer::new(move || {
        App::new()
            .app_data(web::Data::new(tasks_clone2.clone()))
            .route("/execute_workflows", web::post().to(execute_workflows))
    })
    .bind(("127.0.0.1", 8086))?
    .run()
    .await
}
3️⃣ How It Works
Tasks with dependencies are added via RabbitMQ queue workflow_tasks.

Each task can have max_retries if it fails.

Workflow execution endpoint (/execute_workflows) checks dependencies:

Only executes tasks whose dependencies are completed.

Updates status (InProgress, Completed, Failed).

Retries are automatic up to max_retries.

Can be extended to push notifications to WebSocket/SSE dashboards for live monitoring.
